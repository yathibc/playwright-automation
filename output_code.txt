package org.example;

import com.bookings.utils.MailUtils;
import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.chrome.ChromeOptions;
import org.openqa.selenium.support.ui.ExpectedConditions;
import org.openqa.selenium.support.ui.WebDriverWait;
import org.testng.annotations.BeforeMethod;
import org.testng.annotations.Test;

import java.time.Duration;
import java.util.*;

public class TestRCBCSKTickets {

    static WebDriver driver;
    static boolean headless = false;
    static List<String> stands = List.of(
            "PUMA B STAND"
//            ,"BOAT C STAND"
            ,"QATAR AIRWAYS RAHUL DRAVID PLATINUM LOUNGE"
            );
    static String url = "https://shop.royalchallengers.com/ticket/5";

    @BeforeMethod
    public void setup() throws Exception {
        ChromeOptions options = new ChromeOptions();
        if (headless) {
            options.addArguments("--headless=new");
            options.addArguments("--disable-gpu");
        }
        // Path to Chrome User Data
        String userDataDir = "C:\\Chrome_Profile";
//        userDataDir = "C:\\Users\\yathi\\AppData\\Local\\Google\\Chrome\\User Data\\";
        String profileDir = "Profile 1";
        options.addArguments("user-data-dir=" + userDataDir);
        options.addArguments("profile-directory=" + profileDir);  // Use correct profile6
        // Prevent automatic logout
        options.addArguments("--disable-blink-features=AutomationControlled");
        options.setExperimentalOption("excludeSwitches", new String[]{"enable-automation"});
        options.setExperimentalOption("useAutomationExtension", false);
        driver = new ChromeDriver(options);
//        driver.get("https://shop.royalchallengers.com/ticket");
//        driver.navigate().to(url);
        driver.get(url);
        driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(15));
        driver.manage().timeouts().pageLoadTimeout(Duration.ofSeconds(15));
        driver.manage().window().maximize();
    }
    @Test
    public void RCBTicketsTest() throws Exception {
        try {
            while (true) {
                driver.findElement(By.xpath("//button/../*[contains(text(),'Continue')]")).click();

                List<WebElement> categories = driver.findElements(By.xpath("//p[text()='CATEGORY']/following-sibling::div[1]/div/div/p[1]"));
                for (int i = categories.size() - 1; i >= 0; i--) {
                    if (stands.contains(categories.get(i).getText())) {
                        categories.get(i).click();
                    }
                }

                driver.findElement(By.xpath("//p[text()='How many tickets?']/following-sibling::div[2]/button[text()=2]")).click();
                driver.findElement(By.xpath("//button[text()='Continue']")).click();

                try {
                    List<WebElement> seatElements = driver.findElements(By.cssSelector(".seat[fill='#ffffff']"));

                    // Map<y, Map<seatId, WebElement>>
                    Map<Integer, TreeMap<Integer, WebElement>> rowSeatMap = new LinkedHashMap<>();

                    for (WebElement seat : seatElements) {
                        String id = seat.getAttribute("id"); // e.g., "seat126003"
                        String yStr = seat.getAttribute("y");

                        if (id != null && id.startsWith("seat") && yStr != null) {
                            int seatNumber = Integer.parseInt(id.replace("seat", ""));
                            int y = Integer.parseInt(yStr);

                            rowSeatMap.putIfAbsent(y, new TreeMap<>());
                            rowSeatMap.get(y).put(seatNumber, seat);
                        }
                    }

                    boolean found = false;
                    for (Map.Entry<Integer, TreeMap<Integer, WebElement>> rowEntry : rowSeatMap.entrySet()) {
                        TreeMap<Integer, WebElement> seatsInRow = rowEntry.getValue();
                        Integer prevSeatNum = null;

                        for (Integer seatNum : seatsInRow.keySet()) {
                            if (prevSeatNum != null && seatNum == prevSeatNum + 1) {
                                // Found 2 consecutive seats in the same row
                                seatsInRow.get(prevSeatNum).click();
                                seatsInRow.get(seatNum).click();
                                found = true;
                                break;
                            }
                            prevSeatNum = seatNum;
                        }

                        if (found) break;
                    }

                    if (found) {
                        driver.findElement(By.xpath("//button[contains(text(),'Proceed')]")).click();
                    } else {
                        System.out.println("No consecutive seats found in the same row.");
                    }

                } catch (Exception e) {
                    e.printStackTrace();
                }

                boolean isNotInCheckout = !driver.findElement(By.xpath("//p[text()='Checkout']")).isDisplayed();
                if (isNotInCheckout) {
                    driver.navigate().to(url);
                } else {
                    try {
                        Thread.sleep(5000);
                        WebElement timingAlert = driver.findElement(By.xpath("//div[contains(text(),'minutes to complete your booking.')]/following-sibling::footer//button[contains(text(),'Continue')]"));
                        WebDriverWait wait = new WebDriverWait(driver, Duration.ofSeconds(15));
                        wait.until(ExpectedConditions.visibilityOf(timingAlert));
                        timingAlert.click();
                    }catch (Exception e){e.printStackTrace();};
                    driver.findElement(By.xpath("//button[contains(text(),'PAY NOW')]")).click();
                    break;
                }

            }

            MailUtils.senMail("RCB Tickets added to Cart", "https://shop.royalchallengers.com/checkout");

        } catch (Exception e) {
//            driver.close();
            e.printStackTrace();
        }

    }
}
